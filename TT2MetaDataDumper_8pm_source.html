31 package Bio::EnsEMBL::MetaData::MetaDataDumper::TT2MetaDataDumper;
32 use base qw( Bio::EnsEMBL::MetaData::MetaDataDumper );
33 use Bio::EnsEMBL::Utils::Argument qw(rearrange);
34 use Carp;
35 use XML::Simple;
36 use strict;
37 use warnings;

39 sub new {
40  my ($proto, @args) = @_;
41  my $self = $proto->SUPER::new(@args);
42  $self->{file} ||= 'species.tt2';
43  $self->{division} ||= 'Ensembl';
44  $self->{div_links} = {
45  EnsemblBacteria => "http://bacteria.ensembl.org/",
46  EnsemblFungi => "http://fungi.ensembl.org/",
47  EnsemblMetazoa => "http://metazoa.ensembl.org/",
48  EnsemblProtists => "http://protists.ensembl.org/",
49  EnsemblPlants => "http://plants.ensembl.org/",
50  Ensembl => "http://www.ensembl.org/",};
51  $self->{div_names} = {EnsemblBacteria => "Bacteria",
52  EnsemblFungi => "Fungi",
53  EnsemblMetazoa => "Metazoa",
54  EnsemblProtists => "Protists",
55  EnsemblPlants => "Plants",
56  Ensembl => "Ensembl",};
57 
58  return $self;
59 }
60 
61 sub do_dump {
62  my ($self, $metadata, $outfile) = @_;
63  my $eg_version = $metadata->{genome}->[0]->{dbname};
64  $eg_version =~ s/[a-z]+_([0-9]+)_[0-9]+_[0-9]+/$1/;
65  my $td = '<td>';
66  my $tdx = '</td>';
67  my $tr = '<tr>';
68  my $trx = '</tr>';
69  open(my $tt2_file, '>', $outfile) ||
70  croak "Could not write to " . $outfile;
71  print $tt2_file <<"ENDHEAD";
72  <style type="text/css" title="currentStyle">
73  \@import "/static/css/grid.css";
74  </style>
75 
76  <script type="text/javascript" charset="utf-8" src="/static/js/jquery.dataTables.min.js"></script>
77 
78  <script type="text/javascript">
79  $(document).ready( function () {
80  $('#species_list').dataTable({
81  "sPaginationType": "full_numbers",
82  "iDisplayLength": 25,
83  "aLengthMenu": [[25, 50, 100, -1], [25, 50, 100, "All"]]
84 });
85 } );
86 </script>
87 <div id="static">
88 <h2>Ensembl Genomes Release $eg_version</h2>
89 <p>Ensembl Genomes species metadata are also available via FTP: <a href="ftp://ftp.ensemblgenomes.org/pub/current/species.txt">Tabular text</a> | <a href="ftp://ftp.ensemblgenomes.org/pub/current/species_metadata.json">JSON</a> | <a href="ftp://ftp.ensemblgenomes.org/pub/current/species_metadata.xml">XML</a></p>
90 <table cellpadding="0" cellspacing="0" border="1px dotted #eeeeee"
91 class="display" id="species_list">
92 <thead>
93  <tr>
94  <th>Species</th><th>Division</th><th>Taxonomy ID</th><th>Assembly</th><th>Genebuild</th><th>Variation</th><th>Pan compara</th><th>Genome alignments</th><th>Other alignments</th>
95  </tr>
96  </thead>
97  <tbody>
98 ENDHEAD
99 
100  for my $md (@{$metadata->{genome}}) {
101  my $div_link = $self->{div_links}->{$md->{division}};
102  my $div_name = $self->{div_names}->{$md->{division}};
103  if (!defined $div_link) {
104  croak "No division defined for $md->{name}";
105  }
106  $div_link .= $md->{species};
107  print $tt2_file join("",
108  ($tr,
109  $td,
110  "<a href='${div_link}'>",
111  $md->{name},
112  '</a>',
113  $tdx,
114  $td,
115  "<a href='",
116  $div_link,
117  "'>",
118  $div_name,
119  '</a>',
120  $td,
121  "<a href='http://www.uniprot.org/taxonomy/" .
122  $md->{taxonomy_id} .
123  "'>" . $md->{taxonomy_id} . "</a>",
124  $tdx,
125  $td,
126  $md->{assembly_name},
127  $tdx,
128  $td,
129  $md->{genebuild},
130  $tdx,
131  $td,
132  $self->yesno($self->count_variation($md)),
133  $tdx,
134  $td,
135  $self->yesno($md->{pan_species}),
136  $tdx,
137  $td,
138  $self->yesno($self->count_dna_compara($md)),
139  $tdx,
140  $td,
141  $self->yesno($self->count_alignments($md)),
142  $tdx,
143  $trx,
144  "\n"));
145  } ## end for my $md (@{$metadata...})
146  print $tt2_file <<'ENDFOOT';
147 </tbody>
148 <tfoot>
149 <tr>
150  <th>Species</th><th>Division</th><th>Taxonomy ID</th><th>Assembly</th><th>Genebuild</th><th>Variation</th><th>Pan compara</th><th>Genome alignments</th><th>Other alignments</th>
151 </tr>
152 </tfoot>
153 </table>
154 </div>
155 ENDFOOT
156  close $tt2_file;
157  return;
158 } ## end sub do_dump

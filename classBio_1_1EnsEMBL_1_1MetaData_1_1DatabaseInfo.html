sub _load_child {
  my ( $self, $key, $method ) = @_;
  if ( !defined $self->{$key} &&
       defined $self->dbID() &&
       defined $self->adaptor() )
  {
    $self->adaptor()->$method($self);
  }
  return;
}

sub _parse_type {
  my ($dbname) = @_;
  if ( $dbname =~ m/mart/ ) {
    return 'mart';
  }
  else {
    $dbname =~ m/^.+_([a-z]+)_[0-9]+(?:_[a-z0-9]+)?_[0-9a-z]+/;

    if ( defined $1 ) {
      return $1;
    }
    else {
      return 'other';
    }
  }
}

sub count_hash_lengths {
  my ( $self, $hash ) = @_;
  my $tot = 0;
  if ( defined $hash ) {
    for my $v ( values %{$hash} ) {
      $tot += scalar(@$v);
    }
  }
  return $tot;
}

sub dbname {
  my ( $self, $dbname ) = @_;
  if ( defined $dbname ) {
    $self->{dbname} = $dbname;
    if ( !defined $self->{type} ) {
      $self->{type} = _parse_type( $self->dbname() );
    }
  }
  return $self->{dbname};
}

sub new {
  my ( $class, @args ) = @_;
  my $self = $class->SUPER::new(@args);
  my $subject;
  ( $self->{type}, $self->{division}, $self->{dbname}, $subject,
    $self->{species_id} )
    = rearrange( [ 'TYPE', 'DIVISION', 'DBNAME', 'SUBJECT', 'SPECIES_ID' ],
                 @args );
  if ( !defined $self->{type} ) {
    $self->{type} = _parse_type( $self->dbname() );
  }
  $self->{species_id} ||= 1;
  $self->{subject} = $subject;
  return $self;
}

sub subject {
  my ( $self, $subject ) = @_;
  if ( defined $subject ) {
    if ( !$subject->isa("Bio::EnsEMBL::MetaData::GenomeInfo") &&
         !$subject->isa("Bio::EnsEMBL::MetaData::DataReleaseInfo") )
    {
      throw
"The subject of a DatabaseInfo object must be GenomeInfo or DataReleaseInfo";
    }
    $self->{subject} = $subject;
  }
  return $self->{subject};
}

00053 =head1 DESCRIPTION
00054 
00055 Base class for rendering details from an instance of Bio::EnsEMBL::MetaData::GenomeInfo. 
00056 Designed for dumping to multiple per-division files in parallel including a common "all" 
00057 file, (using start, write_metadata and end) or for one-off use (using dump_metadata).

00106 sub dump_metadata {
00107   my ( $self, $metadata, $file, $divisions, $dump_all ) = @_;
00108   # start
00109   $self->start( $file, $divisions, $dump_all );
00110   # iterate
00111   for my $md (@$metadata) {
00112     if ( scalar(@$divisions) > 1 ) {
00113       $self->write_metadata( $md, $self->{all} );
00114     }
00115     $self->write_metadata( $md, $md->{division} );
00116   }
00117   # end
00118   $self->end();
00119   return;

00155 sub start {
00156   my ( $self, $divisions, $file, $dump_all ) = @_;
00157   $self->{files}     = {};
00158   $self->{filenames} = {};
00159   $self->logger()->debug("Opening output files");
00160   for my $division ( @{$divisions} ) {
00161     ( my $out_file = $file ) =~ s/(.+)(\.[^.]+)$/$1_$division$2/;
00162     my $fh;
00163     $self->logger()
00164       ->debug("Opening output file $out_file for division $division");
00165     open( $fh, '>', $out_file ) || croak "Could not open $out_file for writing";
00166     $self->{files}->{$division}     = $fh;
00167     $self->{filenames}->{$division} = $out_file;
00168     $self->{count}{$division}       = 0;
00169   }
00170   if ( defined $dump_all && $dump_all == 1 ) {
00171     my $fh;
00172     $self->logger()->debug("Opening output file $file");
00173     open( $fh, '>', $file ) || croak "Could not open $file for writing";
00174     $self->{files}->{ $self->{all} }     = $fh;
00175     $self->{filenames}->{ $self->{all} } = $file;
00176     $self->{count}{ $self->{all} }       = 0;
00177   }
00178   $self->{files_handles} = {};
00179   $self->logger()
00180     ->debug(
00181            "Opened " . scalar( values %{ $self->{files} } ) . " output files" );
00182   return;
00183 } ## end sub start

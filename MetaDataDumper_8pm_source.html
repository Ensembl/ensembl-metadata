00041 sub new {
00042   my ($proto, @args) = @_;
00043   my $class = ref($proto) || $proto;
00044   my $self = bless({}, $class);
00045   $self->{logger} = get_logger();
00046   ($self->{file}) = rearrange(['FILE'], @args);
00047   $self->{all} = 'all';
00048   return $self;
00049 }

00051 sub start {
00052   my ($self, $divisions, $file, $dump_all) = @_;
00053   $self->{files}     = {};
00054   $self->{filenames} = {};
00055   $self->logger()->debug("Opening output files");
00056   for my $division (@{$divisions}) {
00057     (my $out_file = $file) =~ s/(.+)(\.[^.]+)$/$1_$division$2/;
00058     my $fh;
00059     $self->logger()
00060       ->debug("Opening output file $out_file for division $division");
00061     open($fh, '>', $out_file) ||
00062       croak "Could not open $out_file for writing";
00063     $self->{files}->{$division}     = $fh;
00064     $self->{filenames}->{$division} = $out_file;
00065     $self->{count}{$division}     = 0;
00066   }
00067   if (defined $dump_all && $dump_all==1) {
00068     my $fh;
00069     $self->logger()->debug("Opening output file $file");
00070     open($fh, '>', $file) || croak "Could not open $file for writing";
00071     $self->{files}->{$self->{all}}     = $fh;
00072     $self->{filenames}->{$self->{all}} = $file;
00073     $self->{count}{$self->{all}}     = 0;
00074   }
00075   $self->{files_handles} = {};
00076   $self->logger()
00077     ->debug(
00078         "Opened " . scalar(values %{$self->{files}}) . " output files");
00079   return;
00080 } ## end sub start
00081 
00082 sub end {
00083   my ($self) = @_;
00084   $self->logger()->debug("Closing all file handles");
00085   for my $fh (values %{$self->{files}}) {
00086     $self->logger()->debug("Closing file handle");
00087     close($fh) || cluck "Could not close file handle for writing";
00088   }
00089   $self->logger()
00090     ->debug(
00091         "Closed " . scalar(values %{$self->{files}}) . " file handles");
00092   return;
00093 }

00095 sub write_metadata {
00096   my ($self, $metadata, $division) = @_;
00097   my $fh = $self->{files}{$division};
00098   if (defined $fh) {
00099     $self->_write_metadata_to_file($metadata, $fh, $self->{count}->{$division});
00100     $self->{count}->{$division} += 1;
00101   }
00102   return;
00103 }

00105 sub _write_metadata_to_file {
00106   my ($self, $metadata, $fh) = @_;
00107   throw "Unimplemented subroutine do_dump() in " .
00108     ref($self) . ". Please implement";
00109   return;
00110 }

00112 sub dump_metadata {
00113   my ($self, $metadata, $file, $divisions, $dump_all) = @_;
00114   # start
00115   $self->start($file, $divisions, $dump_all);
00116   # iterate
00117   for my $md (@$metadata) {
00118     if (scalar(@$divisions) > 1) {
00119       $self->write_metadata($md, $self->{all});
00120     }
00121     $self->write_metadata($md, $md->{division});
00122   }
00123   # end
00124   $self->end();
00125   return;

00127 }

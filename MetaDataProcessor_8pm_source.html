00048 sub process_metadata {
00049   my ( $self, $dbas ) = @_;
00050 
00051   # 1. create hash of DBAs
00052   my $dba_hash = {};
00053   my $comparas = [];
00054   for my $dba ( grep { $_->dbc()->dbname() !~ /ancestral/ } @{$dbas} ) {
00055     my $type;
00056     my $species = $dba->species();
00057     for my $t (qw(core otherfeatures variation funcgen)) {
00058       if ( $dba->dbc()->dbname() =~ m/$t/ ) {
00059         $type = $t;
00060         last;
00061       }
00062     }
00063     if ( defined $type ) {
00064       $dba_hash->{$species}{$type} = $dba if defined $type;
00065     }
00066     elsif ( $dba->dbc()->dbname() =~ m/_compara_/ ) {
00067       push @$comparas, $dba;
00068     }
00069   }
00070 
00071   # 2. iterate through each genome
00072   my $genome_infos = {};
00073   my $n            = 0;
00074   my $total        = scalar( keys %$dba_hash );
00075   while ( my ( $genome, $dbas ) = each %$dba_hash ) {
00076     $self->{logger}->info( "Processing " . $genome . " (" . ++$n . "/$total)" );
00077     $genome_infos->{$genome} = $self->process_genome($dbas);
00078   }
00079 
00080   # 3. apply compara
00081   for my $compara (@$comparas) {
00082     $self->process_compara( $compara, $genome_infos );
00083   }
00084 
00085   return [ values(%$genome_infos) ];

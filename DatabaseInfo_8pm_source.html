00073 sub new {
00074   my ( $class, @args ) = @_;
00075   my $self = $class->SUPER::new(@args);
00076   my $subject;
00077   ( $self->{type}, $self->{division}, $self->{dbname}, $subject,
00078     $self->{species_id} )
00079     = rearrange( [ 'TYPE', 'DIVISION', 'DBNAME', 'SUBJECT', 'SPECIES_ID' ],
00080                  @args );
00081   if ( !defined $self->{type} ) {
00082     $self->{type} = _parse_type( $self->dbname() );
00083   }
00084   $self->{species_id} ||= 1;
00085   $self->{subject} = $subject;
00086   return $self;
00087 }

00164 sub subject {
00165   my ( $self, $subject ) = @_;
00166   if ( defined $subject ) {
00167     if ( !$subject->isa("Bio::EnsEMBL::MetaData::GenomeInfo") &&
00168          !$subject->isa("Bio::EnsEMBL::MetaData::DataReleaseInfo") )
00169     {
00170       throw
00171 "The subject of a DatabaseInfo object must be GenomeInfo or DataReleaseInfo";
00172     }
00173     $self->{subject} = $subject;

00175   return $self->{subject};

00214 sub _parse_type {
00215   my ($dbname) = @_;
00216   if ( $dbname =~ m/mart/ ) {
00217     return 'mart';
00218   }
00219   else {
00220     $dbname =~ m/^.+_([a-z]+)_[0-9]+(?:_[a-z0-9]+)?_[0-9a-z]+/;
00221 
00222     if ( defined $1 ) {
00223       return $1;
00224     }
00225     else {
00226       return 'other';
00227     }
00228   }

00063 sub new {
00064   my ( $class, @args ) = @_;
00065   my $self = $class->SUPER::new(@args);
00066   ( $self->{type}, $self->{division}, $self->{dbname}, $self->{subject}, $self->{species_id} ) =
00067     rearrange( [ 'TYPE', 'DIVISION', 'DBNAME', 'SUBJECT', 'SPECIES_ID' ], @args );
00068   if ( !defined $self->{type} ) {
00069     $self->{type} = _parse_type( $self->dbname() );
00070   }
00071   $self->{species_id} ||= 1;
00072   return $self;

00075 sub _parse_type {
00076   my ($dbname) = @_;
00077   if ( $dbname =~ m/mart/ ) {
00078     return 'mart';
00079   }
00080   else {
00081     $dbname =~ m/^.+_([a-z]+)_[0-9]+(?:_[a-z0-9]+)?_[0-9a-z]+/;
00082 
00083     if ( defined $1 ) {
00084       return $1;
00085     }
00086     else {
00087       return 'other';
00088     }
00089   }
00090 }

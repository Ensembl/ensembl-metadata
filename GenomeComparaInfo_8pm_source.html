59  Arg [-DBNAME] :
60  string - name of the compara database in which the analysis can be found

62  string - optional name for the analysis
63  Arg [-GENOMES] :
64  arrayref - list of genomes involved in analysis
65 
66  Example : $info = Bio::EnsEMBL::MetaData::GenomeComparaInfo->new(...);
67  Description: Creates a new info object

69  Exceptions : none
70  Caller : general
71  Status : Stable
72

75 sub new {
76  my ( $class, @args ) = @_;
77  my $self = $class->SUPER::new(@args);
78  ( $self->{division}, $self->{method}, $self->{dbname},
79  $self->{set_name}, $self->{genomes}, $self->{release} )
80  = rearrange(
81  [ 'DIVISION', 'METHOD', 'DBNAME', 'SET_NAME', 'GENOMES', 'DATA_RELEASE'
82  ],
83  @args );
84  return $self;
85 }

164 =head1 UTILITY METHODS
165 =head2 is_pan_compara
166  Arg : (optional) 1/0 to set
167  Description: Boolean-style method, returns 1 if analysis is part of the pan compara, 0 if not
168  Returntype : 1 or 0
169  Exceptions : none
170  Caller : general
171  Status : Stable

174 sub is_pan_compara {
175  my ($self) = @_;
176  return $self->{division} eq 'EnsemblPan' ? 1 : 0;
177 }
178 
179 =head2 is_peptide_compara
180  Arg : (optional) 1/0 to set
181  Description: Boolean-style method, returns 1 if analysis is part of a peptide compara, 0 if not
182  Returntype : 1 or 0
183  Exceptions : none
184  Caller : general
185  Status : Stable
186 =cut
187 
188 sub is_peptide_compara {
189  my ($self) = @_;
190  return ( $self->{division} ne 'EnsemblPan' &&
191  $self->{method} eq 'PROTEIN_TREES' ) ? 1 : 0;
192 }
193 
194 =head2 is_dna_compara
195  Arg : (optional) 1/0 to set
196  Description: Boolean-style method, returns 1 if analysis is part of a DNA compara, 0 if not
197  Returntype : 1 or 0
198  Exceptions : none
199  Caller : general
200  Status : Stable
201 =cut
202 
203 sub is_dna_compara {
204  my ($self) = @_;
205  return ( $self->{method} eq 'TRANSLATED_BLAT_NET' ||
206  $self->{method} eq 'LASTZ_NET' ||
207  $self->{method} eq 'TBLAT' ||
208  $self->{method} eq 'ATAC' ||
209  $self->{method} eq 'BLASTZ_NET' ) ? 1 : 0;
210 }

226 =head2 to_hash
227  Description: Render compara as plain hash suitable for export as JSON/XML
228  Returntype : Hashref
229  Exceptions : none
230  Caller : general
231  Status : Stable
232 =cut

234 sub to_hash {
235  my ($in) = @_;
236  my $out = { method => $in->{method},
237  division => $in->{division},
238  dbname => $in->{dbname},
239  set_name => $in->{set_name},
240  is_pan_compara => $in->is_pan_compara(),
241  is_peptide_compara => $in->is_peptide_compara(),
242  is_dna_compara => $in->is_dna_compara(), };
243  $out->{genomes} = [];
244  for my $genome ( @{ $in->genomes() } ) {
245  push @{ $out->{genomes} }, $genome->species();
246  }
247  return $out;
248 }

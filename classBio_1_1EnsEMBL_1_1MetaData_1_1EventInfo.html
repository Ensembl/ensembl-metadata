sub _load_child {
  my ( $self, $key, $method ) = @_;
  if ( !defined $self->{$key} &&
       defined $self->dbID() &&
       defined $self->adaptor() )
  {
    $self->adaptor()->$method($self);
  }
  return;
}

sub count_hash_lengths {
  my ( $self, $hash ) = @_;
  my $tot = 0;
  if ( defined $hash ) {
    for my $v ( values %{$hash} ) {
      $tot += scalar(@$v);
    }
  }
  return $tot;
}

sub subject {
  my ( $self, $subject ) = @_;
  if ( defined $subject ) {
    if ( !$subject->isa("Bio::EnsEMBL::MetaData::GenomeInfo") &&
         !$subject->isa("Bio::EnsEMBL::MetaData::DatabaseInfo") &&
         !$subject->isa("Bio::EnsEMBL::MetaData::GenomeComparaInfo") )
    {
      throw "Subject must be GenomeInfo, DatabaseInfo or GenomeComparaInfo";
    }
    $self->{subject} = $subject;
  }
  return $self->{subject};
}

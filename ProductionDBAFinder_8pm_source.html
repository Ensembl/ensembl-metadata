00042 sub new {
00043   my ( $proto, @args ) = @_;
00044   my $self = $proto->SUPER::new(@args);
00045   # get the production database
00046   my ( $mhost, $mport, $muser, $mpass, $mdbname ) =
00047     rearrange( [ 'MHOST', 'MPORT', 'MUSER', 'MPASS', 'MDBNAME' ],
00048                @args );
00049   $self->{production_dbc} =
00050     Bio::EnsEMBL::DBSQL::DBConnection->new( -USER =>,
00051                                             $muser,
00052                                             -PASS =>,
00053                                             $mpass,
00054                                             -HOST =>,
00055                                             $mhost,
00056                                             -PORT =>,
00057                                             $mport,
00058                                             -DBNAME =>,
00059                                             $mdbname );
00060 
00061   return $self;
00062 }

00064 sub get_dbas {
00065   my ($self) = @_;
00066   if ( !defined $self->{dbas} ) {
00067     $self->{dbas} = Bio::EnsEMBL::Registry->get_all_DBAdaptors();
00068     $self->{logger}
00069       ->info( "Found " . scalar( @{ $self->{dbas} } ) . " DBAs" );
00070     # get parents and hash by DB
00071     my $dbs = {};
00072     for my $dba ( @{ $self->{dbas} } ) {
00073       push @{ $dbs->{ $dba->dbc()->dbname() } }, $dba;
00074     }
00075 
00076     my @dbs;
00077     # restrict by division
00078     if ( defined $self->{division} ) {
00079       @dbs = @{
00080         $self->{production_dbc}->sql_helper()->execute_simple(
00081           -SQL =>
00082 q/select full_db_name as db_name from db_list join db using (db_id) 
00083 join species using (species_id) join division_species using (species_id) 
00084 join division using (division_id)  where db.is_current=1 and division.name=? 
00085 UNION select db_name from division_db join division using (division_id) 
00086 where division_db.is_current=1 and db_type='COMPARA' and division.name=?/,
00087           -PARAMS => [ $self->{division}, $self->{division} ] ) };
00088     }
00089     elsif ( defined $self->{species} ) {
00090       @dbs = @{
00091         $self->{production_dbc}->sql_helper()->execute_simple(
00092           -SQL =>
00093 q/select full_db_name as db_name from db_list join db using (db_id) join species using (species_id) where db.is_current=1 and production_name=?/,
00094           -PARAMS => [ $self->{species} ] ) };
00095     }
00096     else {
00097       @dbs = @{
00098         $self->{production_dbc}->sql_helper()->execute_simple(
00099           -SQL =>
00100 q/select full_db_name as db_name from db_list join db using (db_id) where db.is_current=1 
00101 UNION select db_name from division_db where division_db.is_current=1 and db_type='COMPARA'/
00102         ) };
00103 
00104     }
00105     # filter by pattern/dbname
00106     if ( defined $self->{dbname} ) {
00107       @dbs = grep { $_ eq $self->{dbname} } @dbs;
00108     }
00109     elsif ( defined $self->{pattern} ) {
00110       @dbs = grep { $_ =~ m/$self->{pattern}/i } @dbs;
00111     }
00112 
00113     $self->{dbas} = [];
00114     for my $db (@dbs) {
00115       my $dba_list = $dbs->{$db};
00116       if ( defined $dba_list ) {
00117         push @{ $self->{dbas} }, @$dba_list;
00118       }
00119       else {
00120         throw "Expected database $db not found";
00121       }
00122     }
00123   } ## end if ( !defined $self->{...})
00124   return $self->{dbas};
00125 } ## end sub get_dbas

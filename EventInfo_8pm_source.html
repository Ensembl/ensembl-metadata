00095 sub subject {
00096   my ( $self, $subject ) = @_;
00097   if ( defined $subject ) {
00098     if ( !$subject->isa("Bio::EnsEMBL::MetaData::GenomeInfo") &&
00099          !$subject->isa("Bio::EnsEMBL::MetaData::DatabaseInfo") &&
00100          !$subject->isa("Bio::EnsEMBL::MetaData::GenomeComparaInfo") )
00101     {
00102       throw "Subject must be GenomeInfo, DatabaseInfo or GenomeComparaInfo";
00103     }
00104     $self->{subject} = $subject;
00105   }
00106   return $self->{subject};
00107 }

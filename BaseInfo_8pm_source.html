00071 sub to_hash {
00072   my ( $in, $keen ) = @_;
00073   my $out;
00074   my $type = ref $in;
00075   if ( defined $keen &&
00076        $keen == 1 &&
00077        $type =~ m/Bio::EnsEMBL::MetaData::[A-z]+Info/ )
00078   {
00079     $in->_preload();
00080   }
00081   if ( $type eq 'ARRAY' ) {
00082     $out = [];
00083     for my $item ( @{$in} ) {
00084       push @{$out}, to_hash( $item, $keen );
00085     }
00086   }
00087   elsif ( $type eq 'HASH' || $type eq 'Bio::EnsEMBL::MetaData::GenomeInfo' ) {
00088     $out = {};
00089     while ( my ( $key, $val ) = each %$in ) {
00090       if ( $key ne 'dbID' && $key ne 'adaptor' && $key ne 'logger' ) {
00091 
00092 # deal with keys starting with numbers, which are not valid element names in XML
00093         if ( $key =~ m/^[0-9].*/ ) {
00094           $key = '_' . $key;
00095         }
00096         $out->{$key} = to_hash( $val, $keen );
00097       }
00098 
00099     }
00100   }
00101   elsif ( $type =~ m/Bio::EnsEMBL::MetaData::[A-z]+Info/ ) {
00102     $out = $in->to_hash($keen);
00103   }
00104   else {
00105     $out = $in;
00106   }
00107   if ( defined $keen &&
00108        $keen == 1 &&
00109        $type =~ m/Bio::EnsEMBL::MetaData::[A-z]+Info/ )
00110   {
00111     $in->_preload();
00112   }
00113   return $out;
00114 } ## end sub to_hash

00041 sub new {
00042   my ($proto, @args) = @_;
00043   my $self = $proto->SUPER::new(@args);
00044   $self->{file} ||= 'uniprot_report.txt';
00045   return $self;
00046 }
00047 
00048 sub start {
00049   my ($self, $divisions, $file, $dump_all) = @_;
00050   $self->SUPER::start($divisions, $file, $dump_all);
00051   for my $fh (values %{$self->{files}}) {
00052       print $fh '#'. join("\t",
00053                        qw(name species division taxonomy_id assembly_id assembly_name genebuild nProteinCoding nProteinCodingUniProtKBSwissProt nProteinCodingUniProtKBTrEMBL uniprotCoverage)
00054     ) .
00055     "\n";
00056   }
00057   return;
00058 }

00060 sub _write_metadata_to_file {
00061   my ($self, $md, $fh) = @_; 
00062   print $fh join(
00063                 "\t",
00064                 ($md->name(),
00065                  $md->species(),
00066                  $md->division(),
00067                  $md->taxonomy_id(),
00068                  $md->assembly_id()   || '',
00069                  $md->assembly_name() || '',
00070                  $md->genebuild()     || '',
00071                  $md->annotations()->{nProteinCoding},
00072                  $md->annotations()->{nProteinCodingUniProtKBSwissProt},
00073                  $md->annotations()->{nProteinCodingUniProtKBTrEMBL},
00074                  sprintf("%.2f", $md->get_uniprot_coverage($md)),
00075                  "\n"));
00076   return;
00077 }

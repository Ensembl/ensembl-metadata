00067   my ( $self, $organism ) = @_;

00075 
00076     if ( defined $dbID ) {
00077       $organism->dbID($dbID);
00078       $organism->adaptor($self);
00079     }
00080   }
00081   if ( defined $organism->dbID() ) {
00082     $self->update($organism);
00083   }
00084   else {
00085     $self->dbc()->sql_helper()->execute_update(
00086       -SQL =>
00087         q/insert into organism(name,display_name,strain,serotype,taxonomy_id,
00088 species_taxonomy_id,is_reference)

00090       -PARAMS => [ $organism->name(),        $organism->display_name(),
00091                    $organism->strain(),      $organism->serotype(),
00092                    $organism->taxonomy_id(), $organism->species_taxonomy_id(),
00093                    $organism->is_reference() ],

00095         my ( $sth, $dbh, $rv ) = @_;
00096         $organism->dbID( $dbh->{mysql_insertid} );

00159 sub fetch_all_by_taxonomy_ids {
00160   my ( $self, $ids ) = @_;
00161 
00162   # filter list down
00163   my %ids = map { $_ => 1 } @$ids;
00164   my @tids =
00165     grep { defined $ids{$_} }
00166     @{ $self->dbc()->sql_helper()
00167       ->execute_simple( -SQL => q/select distinct taxonomy_id from organism/ )
00168     };
00169   my @genomes = ();
00170   my $it = natatime 1000, @tids;
00171   while ( my @vals = $it->() ) {
00172     my $sql =
00173       _get_base_sql() . ' where taxonomy_id in (' . join( ',', @vals ) . ')';
00174     @genomes = ( @genomes, @{ $self->_fetch_generic( $sql, [] ) } );
00175   }
00176   return \@genomes;
00177 }

00423         Bio::EnsEMBL::Registry->get_DBAdaptor( "multi", "taxonomy" );
00424       if ( !defined $tax_dba ) {
00425         # can't find, so try and create alongside metadata
00426         my $args = { -USER   => $self->db()->dbc()->user(),
00427                      -PORT   => $self->db()->dbc()->port(),
00428                      -PASS   => $self->db()->dbc()->pass(),
00429                      -HOST   => $self->db()->dbc()->host(),
00430                      -DBNAME => 'ncbi_taxonomy' };
00431         $tax_dba =
00432           Bio::EnsEMBL::Taxonomy::DBSQL::TaxonomyDBAdaptor->new(%$args)
00433           ->(%$args);
00434       }
00435       if ( defined $tax_dba ) {
00436         $self->{taxonomy_adaptor} = $tax_dba->get_TaxonomyNodeAdaptor();
00437       }
00438     }

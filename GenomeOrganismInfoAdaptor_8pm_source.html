00031 =head1 DESCRIPTION
00032 
00033 Adaptor for storing and retrieving GenomeInfo objects from MySQL genome_info database
00034 
00035 To start working with an adaptor:
00036 
00037 # getting an adaptor
00038 ## adaptor for latest public EG release
00039 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->build_eg_adaptor();
00040 ## adaptor for specified public EG release
00041 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->build_eg_adaptor(21);
00042 ## manually specify a given database
00043 my $dbc = Bio::EnsEMBL::DBSQL::DBConnection->new(
00044 -USER=>'anonymous',
00045 -PORT=>4157,
00046 -HOST=>'mysql-eg-publicsql.ebi.ac.uk',
00047 -DBNAME=>'genome_info_21');
00048 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->new(-DBC=>$dbc);
00049 
00050 To find genomes, use the fetch methods e.g.
00051 
00052 # find a genome by name
00053 my $genome = $gdba->fetch_by_name('arabidopsis_thaliana');
00054 
00055 # find and iterate over all genomes
00056 for my $genome (@{$gdba->fetch_all()}) {
00057     print $genome->name()."\n";
00058 }
00059 
00060 # find and iterate over all genomes from plants
00061 for my $genome (@{$gdba->fetch_all_by_division('EnsemblPlants')}) {
00062     print $genome->name()."\n";
00063 }
00064 
00065 # find and iterate over all genomes with variation
00066 for my $genome (@{$gdba->fetch_all_with_variation()}) {
00067     print $genome->name()."\n";
00068 }
00069 
00070 # find all comparas for the division of interest
00071 my $comparas = $gdba->fetch_all_compara_by_division('EnsemblPlants');
00072 
00073 # find the peptide compara
00074 my ($compara) = grep {$_->is_peptide_compara()} @$comparas;
00075 print $compara->division()." ".$compara->method()."(".$compara->dbname().")\n";
00076 
00077 # print out all the genomes in this compara
00078 for my $genome (@{$compara->genomes()}) {
00079     print $genome->name()."\n";
00080 }

47 sub run {
48  my ($self) = @_;
49  my $dbas = {};
50  my $compara_name = $self->param_required('species');
51  $log->info("Processing compara $compara_name");
52 
53  # TODO
54  my $compara_dba = $self->get_DBAdaptor("compara");
55 
56  $log->info("Connecting to info database");
57  my $dba =
58  Bio::EnsEMBL::MetaData::DBSQL::MetaDataDBAdaptor->new(
59  -USER =>,
60  $self->param('info_user'),
61  -PASS =>,
62  $self->param('info_pass'),
63  -HOST =>,
64  $self->param('info_host'),
65  -PORT =>,
66  $self->param('info_port'),
67  -DBNAME =>,
68  $self->param('info_dbname')
69  );
70  my $gdba = $dba->get_GenomeInfoAdaptor();
71  my $cdba = $dba->get_GenomeComparaInfoAdaptor();
72 
73  # set the release to use when storing genomes
74  my $rdba = $dba->get_DataReleaseInfoAdaptor();
75  my $release;
76  if ( defined $self->param('eg_release') ) {
77  $release =
78  $rdba->fetch_by_ensembl_genomes_release(
79  $self->param('eg_release') );
80  }
81  else {
82  $release =
83  $rdba->fetch_by_ensembl_release( $self->param('release') );
84  }
85  $gdba->data_release($release);
86  #$cdba->data_release($release);
87 
88  my $upd = $self->param('force_update') || 0;
89 
90  my $opts = { -INFO_ADAPTOR => $gdba,
91  -ANNOTATION_ANALYZER =>
92  Bio::EnsEMBL::MetaData::AnnotationAnalyzer->new(),
93  -COMPARA => 0,
94  -CONTIGS => 0,
95  -FORCE_UPDATE => $upd,
96  -VARIATION => 0 };
97  $log->info("Processing $compara_name");
98  my $processor =
99  Bio::EnsEMBL::MetaData::MetaDataProcessor->new(%$opts);
100  my $compara_infos = $processor->process_compara( $compara_dba, {} );
101 
102  for my $compara_info (@$compara_infos) {
103  my $nom = $compara_info->method() . "/" . $compara_info->set_name();
104  if ( defined $compara_info->dbID() ) {
105  $log->info( "Updating compara info for " . $nom );
106  $cdba->update($compara_info);
107  }
108  else {
109  $log->info( "Storing compara info for " . $nom );
110  $cdba->store($compara_info);
111  }
112  }
113 
114  $log->info("Completed processing compara $compara_name");
115  return;
116 } ## end sub run

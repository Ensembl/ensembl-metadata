00034 =head1 SYNOPSIS
00035 
00036 # creation from a database server
00037 Bio::EnsEMBL::LookUp::LocalLookUp->register_all_dbs( $conf->{host},
00038        $conf->{port}, $conf->{user}, $conf->{pass}, $conf->{db});

00040 my $dbas = $lookup->registry()->get_all();
00041 $dbas = $lookup->get_all_by_taxon_id(388919);
00042 $dbas = $lookup->get_by_name_pattern("Escherichia.*");
00043 
00044 # creation from a URL
00045 my $lookup = Bio::EnsEMBL::LookUp::::LocalLookUp->new(-URL=>'http://bacteria.ensembl.org/registry.json');
00046 
00047 =head1 DESCRIPTION
00048 
00049 This module is an implementation of Bio::EnsEMBL::LookUp that uses a local hash to store information about available Ensembl Genomes data. 
00050 It can be constructed in a number of different ways:
00051 
00052 A remote cache file containing a JSON representation of the hash can be supplied:
00053 
00054     my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(-URL=>'http://bacteria.ensembl.org/registry.json');
00055 
00056 Alternatively, a local file containing the required JSON can be specified instead:
00057 
00058     my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(-FILE=>"/path/to/reg.json");
00059 
00060 Finally, a Registry already loaded with core databases can be supplied:
00061 
00062     my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(-REGISTRY=>'Bio::EnsEMBL::Registry');
00063 
00064 If the standard Registry is used, the argument can be omitted completely:
00065 
00066     my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new();
00067 
00068 To populate the registry with just the Ensembl Genomes databases for the current software 
00069 release on a specified server, the following method can be used:
00070 
00071     Bio::EnsEMBL::LookUp::LocalLookUp->register_all_dbs( $host,
00072        $port, $user, $pass);
00073 
00074 Once a lookup has been created, it can be used as specified in Bio::EnsEMBL::LookUp

00344 sub _hash_dba_from_values {
00345   my ( $self, $dba, $taxids, $aliases, $accessions, $vgc ) = @_;
00346 
00347   for my $taxid ( @{$taxids} ) {
00348     if ( defined $taxid ) {
00349       push @{ $self->{dbas_by_taxid}{$taxid} }, $dba;
00350     }
00351   }
00352   for my $name ( uniq( @{$aliases} ) ) {
00353     push @{ $self->{dbas_by_name}{$name} }, $dba;
00354   }
00355   for my $acc ( uniq( @{$accessions} ) ) {
00356     push @{ $self->{dbas_by_vacc}{$acc} }, $dba;
00357     $acc =~ s/\.[0-9]+$//x;
00358     push @{ $self->{dbas_by_acc}{$acc} }, $dba;
00359   }
00360   if ( defined $vgc ) {
00361     if ( defined $self->{dbas_by_vgc}{$vgc} ) {
00362       croak "DBA with GC accession $vgc already registered";
00363     }
00364     $self->{dbas_by_vgc}{$vgc} = $dba;
00365     $vgc =~ s/\.[0-9]+$//x;
00366     if ( defined $self->{dbas_by_gc}{$vgc} ) {
00367       croak "DBA with GC set chain $vgc already registered";
00368     }
00369     $self->{dbas_by_gc}{$vgc} = $dba;
00370   }
00371   $self->{dbas}{ $self->_dba_id($dba) } = $dba;
00372   return;

00395 sub _registry_to_hash {
00396   my ($self) = @_;
00397   # hash dbcs and dbas by locators
00398   my $dbc_hash;
00399   my $dba_hash;
00400   for my $dba ( values %{ $self->{dbas} } ) {
00401     my $dbc_loc = _dbc_to_locator( $dba->dbc() );
00402     $dbc_hash->{$dbc_loc} = $dba->dbc();
00403     push @{ $dba_hash->{$dbc_loc} }, $dba;
00404   }
00405   # create auxillary hashes
00406   my $acc_hash   = _invert_dba_hash( $self->{dbas_by_vacc} );
00407   my $name_hash  = _invert_dba_hash( $self->{dbas_by_name} );
00408   my $taxid_hash = _invert_dba_hash( $self->{dbas_by_taxid} );
00409   my $gc_hash    = _invert_dba_hash( $self->{dbas_by_vgc} );
00410   # create array of hashes
00411   my $out_arr;
00412   while ( my ( $dbc_loc, $dbconn ) = each( %{$dbc_hash} ) ) {
00413     my $dbc_h = { driver   => $dbconn->driver(),
00414                   host     => $dbconn->host(),
00415                   port     => $dbconn->port(),
00416                   username => $dbconn->username(),
00417                   password => $dbconn->password(),
00418                   dbname   => $dbconn->dbname(),
00419                   species  => [] };
00420     for my $dba ( @{ $dba_hash->{$dbc_loc} } ) {
00421       my $dba_loc = _dba_to_locator($dba);
00422       my $gc      = $gc_hash->{$dba_loc};
00423       if ( ref $gc eq 'ARRAY' ) {
00424         $gc = $gc->[0];
00425       }
00426       push @{ $dbc_h->{species} },
00427         { species_id => $dba->species_id(),
00428           species    => $dba->species(),
00429           taxids     => $taxid_hash->{$dba_loc},
00430           aliases    => $name_hash->{$dba_loc},
00431           accessions => $acc_hash->{$dba_loc},
00432           gc         => $gc };
00433     }
00434     push @{$out_arr}, $dbc_h;
00435   }
00436   return $out_arr;
00437 } ## end sub _registry_to_hash

00848 =head2 register_all_dbs

00850     Argument    : Host

00859 sub register_all_dbs {
00860   my ( $class, $host, $port, $user, $pass, $regexp ) = @_;
00861 
00862   if ( !$regexp ) {
00863     $regexp = '_core_[0-9]+_' . software_version() . '_[0-9]+';

00959 =head2 _query_multispecies_db
00960     Description : Find all species in the multispecies database
00961 =cut
00962 
00963 sub _query_multispecies_db {
00964   my ( $species, %db_args ) = @_;
00965   my @species_array;
00966   eval {
00967     my $dbh = Bio::EnsEMBL::DBSQL::DBConnection->new(%db_args);
00968     my $sql;
00969     my @params = ('species.db_name');
00970 
00971     if ( defined $species ) {
00972       $sql =
00973 'select meta_value, species_id from meta where meta_key =? and meta_value =?';
00974       push( @params, $species );
00975     }
00976     else {
00977       $sql =
00978 'select distinct meta_value, species_id from meta where meta_key =? and species_id is not null';
00979     }
00980     my $sth = $dbh->prepare($sql);
00981     $sth->execute(@params);
00982     while ( my $row = $sth->fetchrow_arrayref() ) {
00983       my ( $species_name, $species_id ) = @{$row};
00984       push( @species_array,
00985             { '-SPECIES'    => $species_name,
00986               '-SPECIES_ID' => $species_id } );
00987     }
00988     $sth->finish();
00989   };
00990   throw("Error detected: $@") if $@;
00991   return \@species_array;
00992 } ## end sub _query_multispecies_db

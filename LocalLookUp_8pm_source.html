30 =head1 NAME
31 
32 Bio::EnsEMBL::LookUp::LocalLookUp
33 
34 =head1 SYNOPSIS
35 
36 # creation from a database server
37 Bio::EnsEMBL::LookUp::LocalLookUp->register_all_dbs( $conf->{host},
38  $conf->{port}, $conf->{user}, $conf->{pass}, $conf->{db});
39 my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new();
40 my $dbas = $lookup->registry()->get_all();
41 $dbas = $lookup->get_all_by_taxon_id(388919);
42 $dbas = $lookup->get_by_name_pattern("Escherichia.*");
43 
44 # creation from a URL
45 my $lookup = Bio::EnsEMBL::LookUp::::LocalLookUp->new(-URL=>'http://bacteria.ensembl.org/registry.json');
46 
47 =head1 DESCRIPTION
48 
49 This module is an implementation of Bio::EnsEMBL::LookUp that uses a local hash to store information about available Ensembl Genomes data.
50 It can be constructed in a number of different ways:
51 
52 A remote cache file containing a JSON representation of the hash can be supplied:
53 
54  my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(-URL=>'http://bacteria.ensembl.org/registry.json');
55 
56 Alternatively, a local file containing the required JSON can be specified instead:
57 
58  my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(-FILE=>"/path/to/reg.json");
59 
60 Finally, a Registry already loaded with core databases can be supplied:
61 
62  my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(-REGISTRY=>'Bio::EnsEMBL::Registry');
63 
64 If the standard Registry is used, the argument can be omitted completely:
65 
66  my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new();
67 
68 To populate the registry with just the Ensembl Genomes databases for the current software
69 release on a specified server, the following method can be used:
70 
71  Bio::EnsEMBL::LookUp::LocalLookUp->register_all_dbs( $host,
72  $port, $user, $pass);
73 
74 Once a lookup has been created, it can be used as specified in Bio::EnsEMBL::LookUp

112 =head2 new
113 
114  Description : Creates a new instance of this object.
115  Arg [-REGISTRY] : (optional) Registry module to use (default is Bio::EnsEMBL::Registry)
116  Arg [-NO_CACHE] : (optional) int 1
117  This option will turn off the use of a local cache file for storing species details
118  Arg [-CACHE_FILE] : (optional) String (default lookup_cache.json)
119  This option allows use of a user specified local cache file
120  Arg [-URL] : (optional)
121  This option allows the use of a remote resource supplying species details in JSON format. This is not used if a local cache exists and -NO_CACHE is not set
122  Arg [-FILE] : (optional)
123  This option allows the use of a file supplying species details in JSON format. This is not used if a local cache exists and -NO_CACHE is not set
124  Returntype : Instance of lookup
125  Status : Stable
126 
127  Example :
128  my $lookup = Bio::EnsEMBL::LookUp::LocalLookUp->new(
129  -REGISTRY => $reg);
130 =cut

132 sub new {
133  my ( $class, @args ) = @_;
134  my $self = bless( {}, ref($class) || $class );
135  my ( $reg, $url, $file, $nocache, $cache_file, $clear_cache, $skip_contigs ) =
136  rearrange( [qw(registry url file no_cache cache_file clear_cache skip_contigs)],
137  @args );
138  if ( !defined $reg ) { $reg = 'Bio::EnsEMBL::Registry' }
139  $self->{registry} = $reg;
140  $self->{dbas_by_taxid} = {};
141  $self->{dbas_by_name} = {};
142  $self->{dbas_by_acc} = {};
143  $self->{dbas_by_vacc} = {};
144  $self->{dbas_by_gc} = {};
145  $self->{dbas_by_vgc} = {};
146  $self->{dbas} = {};
147  $self->{dbc_meta} = {};
148  $self->{skip_contigs} = $skip_contigs;
149  $cache_file ||= $default_cache_file;
150  $self->cache_file($cache_file);
151 
152  if ( defined $clear_cache && $clear_cache == 1 ) {
153  $self->clear_cache();
154  }
155  if ( !$nocache && !defined $file && -e $cache_file ) {
156  $file = $cache_file;
157  }
158  if ( defined $file ) {
159  $self->load_registry_from_file($file);
160  }
161  elsif ( defined $url ) {
162  $self->load_registry_from_url($url);
163  }
164  else {
165  $self->update_from_registry();
166  }
167  if ( !$nocache && !-e $cache_file ) {
168  $self->write_registry_to_file($cache_file);
169  }
170  return $self;
171 } ## end sub new

257 sub _get_dbc_meta {
258  my ( $self, $dbc ) = @_;
259  my $name = $dbc->dbname();
260  if ( !defined $self->{dbc_meta}{$name} ) {
261  # load metadata in batch for the whole database
262  $self->{dbc_meta}{$name} = {};
263  # taxid
264 
265  $dbc->sql_helper()->execute(
266  -SQL =>
267 q/select species_id,meta_value from meta where meta_key='species.taxonomy_id'/,
268  -PARAMS => [],
269  -CALLBACK => sub {
270  my $row = shift @_;
271  $self->{dbc_meta}{$name}{ $row->[0] }{taxid} = $row->[1];
272  } );
273  # aliases
274  $dbc->sql_helper()->execute(
275  -SQL =>
276 'select species_id,meta_value from meta where meta_key=\'species.alias\'',
277  -PARAMS => [],
278  -CALLBACK => sub {
279  my @row = @{ shift @_ };
280  push @{ $self->{dbc_meta}{$name}{ $row[0] }{aliases} }, $row[1];
281  } );
282  # assembly.accession
283  $dbc->sql_helper()->execute(
284  -SQL =>
285 'select species_id,meta_value from meta where meta_key=\'assembly.accession\'',
286  -PARAMS => [],
287  -CALLBACK => sub {
288  my @row = @{ shift @_ };
289  $self->{dbc_meta}{$name}{ $row[0] }{assembly_accession} =
290  $row[1];
291  } );
292 
293  if(!defined $self->{skip_contigs}) {
294  # seq_region_names and synonyms
295  $dbc->sql_helper()->execute(
296  -SQL =>
297  q/select cs.species_id,sr.name,ss.synonym from coord_system cs
298  join seq_region sr using (coord_system_id)
299  left join seq_region_synonym ss using (seq_region_id)/,
300  -PARAMS => [],
301  -CALLBACK => sub {
302  my @row = @{ shift @_ };
303  push @{ $self->{dbc_meta}{$name}{ $row[0] }{accessions} },
304  $row[1];
305  if ( defined $row[2] ) {
306  push @{ $self->{dbc_meta}{$name}{ $row[0] }{accessions} },
307  $row[2];
308  }
309  } );
310  }
311  } ## end if ( !defined $self->{...})
312  return $self->{dbc_meta}{$name};
313 } ## end sub _get_dbc_meta

321 sub _hash_dba {
322  my ( $self, $dba, $update_only ) = @_;
323  my $nom = $self->_dba_id($dba);
324  if ( !defined $update_only || !defined $self->{dbas}{$nom} ) {
325  my $dbc_meta = $self->_get_dbc_meta( $dba->dbc() );
326  my $dba_meta = $dbc_meta->{ $dba->species_id() };
327  $self->_hash_dba_from_values( $dba,
328  [ $dba_meta->{taxid} ],
329  $dba_meta->{aliases},
330  $dba_meta->{accessions},
331  $dba_meta->{assembly_accession} );
332  }
333  return;
334 }

336 =head2 _hash_dba_from_values
337  Description : Add supplied details to the internal hashes of details
338  Argument : Bio::EnsEMBL::DBAdaptor to hash
339  Argument : Arrayref of taxonomy IDs to use as keys
340  Argument : Arrayref of aliases to use as keys
341  Argument : Arrayref of ENA accessions to use as keys
342 =cut

344 sub _hash_dba_from_values {
345  my ( $self, $dba, $taxids, $aliases, $accessions, $vgc ) = @_;
346 
347  for my $taxid ( @{$taxids} ) {
348  if ( defined $taxid ) {
349  push @{ $self->{dbas_by_taxid}{$taxid} }, $dba;
350  }
351  }
352  for my $name ( uniq( @{$aliases} ) ) {
353  push @{ $self->{dbas_by_name}{$name} }, $dba;
354  }
355  for my $acc ( uniq( @{$accessions} ) ) {
356  push @{ $self->{dbas_by_vacc}{$acc} }, $dba;
357  $acc =~ s/\.[0-9]+$//x;
358  push @{ $self->{dbas_by_acc}{$acc} }, $dba;
359  }
360  if ( defined $vgc ) {
361  if ( defined $self->{dbas_by_vgc}{$vgc} ) {
362  croak "DBA with GC accession $vgc already registered";
363  }
364  $self->{dbas_by_vgc}{$vgc} = $dba;
365  $vgc =~ s/\.[0-9]+$//x;
366  if ( defined $self->{dbas_by_gc}{$vgc} ) {
367  croak "DBA with GC set chain $vgc already registered";
368  }
369  $self->{dbas_by_gc}{$vgc} = $dba;
370  }
371  $self->{dbas}{ $self->_dba_id($dba) } = $dba;
372  return;
373 } ## end sub _hash_dba_from_values

375 sub _invert_dba_hash {
376  my ($hash) = @_;
377  my $inv_hash;
378  while ( my ( $item, $dbas ) = each( %{$hash} ) ) {
379  if ( ref $dbas ne 'ARRAY' ) {
380  $dbas = [$dbas];
381  }
382  for my $dba ( @{$dbas} ) {
383  my $dba_loc = _dba_to_locator($dba);
384  push @{ $inv_hash->{$dba_loc} }, $item;
385  }
386  }
387  return $inv_hash;
388 }

390 =head2 _registry_to_hash
391  Description : Generate a hash array structure for the current registry and species details for turning to JSON
392  Returns : Arrayref of hashes

395 sub _registry_to_hash {
396  my ($self) = @_;
397  # hash dbcs and dbas by locators
398  my $dbc_hash;
399  my $dba_hash;
400  for my $dba ( values %{ $self->{dbas} } ) {
401  my $dbc_loc = _dbc_to_locator( $dba->dbc() );
402  $dbc_hash->{$dbc_loc} = $dba->dbc();
403  push @{ $dba_hash->{$dbc_loc} }, $dba;
404  }
405  # create auxillary hashes
406  my $acc_hash = _invert_dba_hash( $self->{dbas_by_vacc} );
407  my $name_hash = _invert_dba_hash( $self->{dbas_by_name} );
408  my $taxid_hash = _invert_dba_hash( $self->{dbas_by_taxid} );
409  my $gc_hash = _invert_dba_hash( $self->{dbas_by_vgc} );
410  # create array of hashes
411  my $out_arr;
412  while ( my ( $dbc_loc, $dbconn ) = each( %{$dbc_hash} ) ) {
413  my $dbc_h = { driver => $dbconn->driver(),
414  host => $dbconn->host(),
415  port => $dbconn->port(),
416  username => $dbconn->username(),
417  password => $dbconn->password(),
418  dbname => $dbconn->dbname(),
419  species => [] };
420  for my $dba ( @{ $dba_hash->{$dbc_loc} } ) {
421  my $dba_loc = _dba_to_locator($dba);
422  my $gc = $gc_hash->{$dba_loc};
423  if ( ref $gc eq 'ARRAY' ) {
424  $gc = $gc->[0];
425  }
426  push @{ $dbc_h->{species} },
427  { species_id => $dba->species_id(),
428  species => $dba->species(),
429  taxids => $taxid_hash->{$dba_loc},
430  aliases => $name_hash->{$dba_loc},
431  accessions => $acc_hash->{$dba_loc},
432  gc => $gc };
433  }
434  push @{$out_arr}, $dbc_h;
435  }
436  return $out_arr;
437 } ## end sub _registry_to_hash

483 sub _load_registry_from_json {
484  my ( $self, $json ) = @_;
485  my $reg_arr = decode_json($json);
486  for my $dbc_h ( @{$reg_arr} ) {
487  my $dbc =
488  Bio::EnsEMBL::DBSQL::DBConnection->new(
489  -driver => $dbc_h->{driver},
490  -host => $dbc_h->{host},
491  -port => $dbc_h->{port},
492  -user => $dbc_h->{username},
493  -pass => $dbc_h->{password},
494  -dbname => $dbc_h->{dbname}
495  );
496  for my $species ( @{ $dbc_h->{species} } ) {
497  my $dba =
498  Bio::EnsEMBL::DBSQL::DBAdaptor->new(
499  -DBCONN => $dbc,
500  -species => $species->{species},
501  -species_id => $species->{species_id},
502  -multispecies_db => 1,
503  -group => 'core', );
504  # no need to register - found automagically
505  my $gc = $species->{gc};
506  if ( ref $gc eq 'ARRAY' ) { $gc = $gc->[0]; }
507  $self->_hash_dba_from_values( $dba,
508  $species->{taxids},
509  $species->{aliases},
510  $species->{accessions},
511  $species->{gc} );
512  }
513  } ## end for my $dbc_h ( @{$reg_arr...})
514  $self->_intern_db_connections();
515  return;
516 } ## end sub _load_registry_from_json

588 =head2 _intern_db_connections()
589  Description : Go through all available DBAdaptors of registry and ensure they use the same

595 sub _intern_db_connections {
596  my ($self) = @_;
597  my $adaptors = $self->registry()->get_all_DBAdaptors();
598  my %dbc_intern;
599  foreach my $dba ( @{$adaptors} ) {
600  my $dbc = $dba->dbc();
601  my $locator = _dbc_to_locator($dbc);
602  if ( exists $dbc_intern{$locator} ) {
603  $dba->dbc( $dbc_intern{$locator} );
604  }
605  else {
606  $dbc_intern{$locator} = $dba->dbc();
607  }
608  }
609  return;
610 }

848 =head2 register_all_dbs
849  Description : Helper method to load the registry with all multispecies core databases on the supplied server
850  Argument : Host
851  Argument : Port
852  Argument : User
853  Argument : Password
854  Argument : (optional) String with database regexp (default is _collection_core_[0-9]_eVersion_[0-9]+)
855  Exceptions : None
856  Return type : None
857 =cut
858 
859 sub register_all_dbs {
860  my ( $class, $host, $port, $user, $pass, $regexp ) = @_;
861 
862  if ( !$regexp ) {
863  $regexp = '_core_[0-9]+_' . software_version() . '_[0-9]+';
864  }
865 
866  if ( !defined $host ) {
867  croak "Host must be supplied for registration of databases";
868  }
869 
870  if ( !defined $port ) {
871  croak
872 "Port must be supplied for registration of databases on host $host";
873  }
874 
875  my $str = "DBI:mysql:host=$host;port=$port";
876  my $dbh = DBI->connect( $str, $user, $pass ) ||
877  croak "Could not connect to database $str (user=$user, pass=$pass)";
878 
879  my @dbnames =
880  grep { m/$regexp/xi }
881  map { $_->[0] } @{ $dbh->selectall_arrayref('SHOW DATABASES') };
882 
883  for my $db (@dbnames) {
884  if ( $db =~ m/.*_collection_core_.*/ ) {
885  _register_multispecies_core( $host, $port, $user, $pass, $db );
886  }
887  else {
888  _register_singlespecies_core( $host, $port, $user, $pass, $db );
889  }
890  }
891  return;
892 } ## end sub register_all_dbs

936 sub _register_multispecies_x {
937

959 =head2 _query_multispecies_db
960  Description : Find all species in the multispecies database
961 =cut
962 
963 sub _query_multispecies_db {
964  my ( $species, %db_args ) = @_;
965  my @species_array;
966  eval {
967  my $dbh = Bio::EnsEMBL::DBSQL::DBConnection->new(%db_args);
968  my $sql;
969  my @params = ('species.db_name');
970 
971  if ( defined $species ) {
972  $sql =
973 'select meta_value, species_id from meta where meta_key =? and meta_value =?';
974  push( @params, $species );
975  }
976  else {
977  $sql =
978 'select distinct meta_value, species_id from meta where meta_key =? and species_id is not null';
979  }
980  my $sth = $dbh->prepare($sql);
981  $sth->execute(@params);
982  while ( my $row = $sth->fetchrow_arrayref() ) {
983  my ( $species_name, $species_id ) = @{$row};
984  push( @species_array,
985  { '-SPECIES' => $species_name,
986  '-SPECIES_ID' => $species_id } );
987  }
988  $sth->finish();
989  };
990  throw("Error detected: $@") if $@;
991  return \@species_array;
992 } ## end sub _query_multispecies_db

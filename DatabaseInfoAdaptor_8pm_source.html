00073 where dbname=? and data_release_id=?/,
00074         -PARAMS =>
00075           [ $data_release_database->dbname(), $data_release_database->subject()->dbID() ] )
00076     };
00077 
00078     if ( defined $dbID ) {
00079       $data_release_database->dbID($dbID);

00083       $self->update($data_release_database);
00084     }

00086       $self->dbc()->sql_helper()->execute_update(
00087         -SQL =>
00088 q/insert into data_release_database(data_release_id,type,division_id,dbname)

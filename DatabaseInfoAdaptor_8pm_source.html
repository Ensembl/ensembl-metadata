00265 sub fetch_databases_DataReleaseInfo {
00266   my ( $self, $release, $division ) = @_;
00267   my $sql = q/select dbname,type,d.name from data_release_database 
00268 join division d using (division_id) where data_release_id=?/;
00269   my $params = [ $release->dbID() ];
00270   if ( defined $division ) {
00271     $sql .= ' and (d.name=? or d.short_name=?)';
00272     $params = [ @$params, $division, $division ];
00273   }
00274 
00275   my $databases = [];
00276 
00277   $self->dbc()->sql_helper()->execute_no_return(
00278     -SQL      => $sql,
00279     -CALLBACK => sub {
00280       my @row = @{ shift @_ };
00281       push @{$databases},
00282         Bio::EnsEMBL::MetaData::DatabaseInfo->new( -SUBJECT  => $release,
00283                                                    -DBNAME   => $row[0],
00284                                                    -TYPE     => $row[1],
00285                                                    -DIVISION => $row[2] );
00286       return;

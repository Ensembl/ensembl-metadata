34 =head1 SYNOPSIS
35 
36 my $adaptor = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->build_ensembl_genomes_adaptor();
37 my $lookup = Bio::EnsEMBL::RemoteLookUp->new(-ADAPTOR=>$adaptor);
38 my $dbas = $lookup->registry()->get_all();
39 $dbas = $lookup->get_all_by_taxon_id(388919);
40 $dbas = $lookup->get_by_name_pattern("Escherichia.*");

42 =head1 DESCRIPTION
43 
44 This module is an implementation of Bio::EnsEMBL::LookUp that uses Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor to
45 access a MySQL database containing information about Ensembl and Ensembl Genomes contents and then instantiate DBAdaptors.
46 
47 To instantiate using the public Ensembl/EG servers for creating DBAdaptors:
48 
49  my $lookup = Bio::EnsEMBL::LookUp::RemoteLookUp->new(-ADAPTOR=>$adaptor);
50 
51 To instantiate to use a specific server on which core databases are located:
52 
53  my $lookup = Bio::EnsEMBL::LookUp::RemoteLookUp->new(-USER=>$user, -HOST=>$host, -PORT=>$port, -ADAPTOR=>$adaptor);
54 
55 Once constructed, the LookUp instance can be used as documented in Bio::EnsEMBL::LookUp.

123 sub _get_args {
124  my ( $self, $genome_info ) = @_;
125  # find arguments applicable to this particular info object
126  my $args = {};
127  if ( !defined $self->{host} ) {
128  # no host specified, so try and use public servers
129  if ( defined $genome_info->data_release()->ensembl_genomes_release() ) {
130  # use EG
131  $args = eg_args();
132  }
133  else {
134  # use Ensembl
135  $args = e_args();
136  }
137  }
138  else {
139  # use host specified
140  $args = { -USER => $self->{user},
141  -PORT => $self->{port},
142  -PASS => $self->{pass},
143  -HOST => $self->{host} };
144  }
145  return $args;
146 }

148 =head2 genome_to_dba
149  Description : Build a Bio::EnsEMBL::DBSQL::DBAdaptor instance with the supplied info object
150  Argument : Bio::EnsEMBL::MetaData::GenomeInfo
151  Exceptions : None
152  Return type : Bio::EnsEMBL::DBSQL::DBAdaptor
153 =cut
154 
155 sub genome_to_dba {
156  my ( $self, $genome_info ) = @_;
157  my $dba;
158  if ( defined $genome_info ) {
159  assert_ref( $genome_info, 'Bio::EnsEMBL::MetaData::GenomeInfo' );
160  $dba = $self->_cache()->{ $genome_info->dbID() };
161  if ( !defined $dba ) {
162 
163  my $args = $self->_get_args($genome_info);
164 
165  $args->{-DBNAME} = $genome_info->dbname();
166  $args->{-SPECIES} = $genome_info->name();
167  $args->{-SPECIES_ID} = $genome_info->species_id();
168  $args->{-MULTISPECIES_DB} =
169  $genome_info->dbname() =~ m/_collection_/ ? 1 : 0;
170  $args->{-GROUP} = 'core';
171 
172  $dba = Bio::EnsEMBL::DBSQL::DBAdaptor->new(%$args);
173 
174  $self->_cache()->{ $genome_info->dbID() } = $dba;
175 
176  }
177  }
178  return $dba;
179 } ## end sub genome_to_dba

181 =head2 genomes_to_dbas
182  Description : Build a set of Bio::EnsEMBL::DBSQL::DBAdaptor instances with the supplied info objects
183  Argument : array ref of Bio::EnsEMBL::MetaData::GenomeInfo
184  Exceptions : None
185  Return type : array ref of Bio::EnsEMBL::DBSQL::DBAdaptor
186 =cut
187 
188 sub genomes_to_dbas {
189  my ( $self, $genomes ) = @_;
190  my $dbas = [];
191  if ( defined $genomes ) {
192  for my $genome ( @{$genomes} ) {
193  push @$dbas, $self->genome_to_dba($genome);
194  }
195  }
196  return $dbas;
197 }

199 =head2 compara_to_dba
200  Description : Build a Bio::EnsEMBL::Compara::DBSQL::DBAdaptor instance with the supplied info object
201  Argument : Bio::EnsEMBL::MetaData::GenomeComparaInfo
202  Exceptions : None

206 sub compara_to_dba {
207  my ( $self, $genome_info ) = @_;
208  assert_ref( $genome_info, 'Bio::EnsEMBL::MetaData::GenomeComparaInfo' );
209  my $dba = $self->_cache()->{compara}->{ $genome_info->dbID() };
210  if ( !defined $dba ) {
211  my $div = $genome_info->division();
212  if ( !$div eq 'Ensembl' ) {
213  $div = 'multi';
214  }
215 
216  my $args = $self->_get_args($genome_info);
217  $args->{-DBNAME} = $genome_info->dbname();
218  $args->{-SPECIES} = $div;
219  $args->{-GROUP} = 'compara';
220  $dba = Bio::EnsEMBL::Compara::DBSQL::DBAdaptor->new(%$args);
221 
222  $self->_cache()->{compara}->{ $genome_info->dbID() } = $dba;
223 
224  }
225  return $dba;
226 }

290 sub get_all_by_accession {
291  my ( $self, $acc ) = @_;
292  my $genomes = $self->{_adaptor}->fetch_all_by_sequence_accession($acc);
293  if ( !defined $genomes || scalar(@$genomes) == 0 ) {

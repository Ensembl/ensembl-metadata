00058 sub do_dump {
00059   my ($self, $metadata, $outfile) = @_;
00060   $self->logger()->info("Writing XML to " . $outfile);
00061   open(my $xml_file, '>', $outfile) ||
00062     croak "Could not write to " . $outfile;
00063   print $xml_file XML::Simple->new()
00064     ->XMLout($self->metadata_to_hash($metadata), RootName => 'genomes');
00065   close $xml_file;
00066   $self->logger()->info("Completed writing XML to " . $outfile);
00067   return;
00068 }

00070 sub start {
00071   my ($self, $divisions, $file, $dump_all) = @_;
00072   $self->SUPER::start($divisions, $file, $dump_all);
00073   for my $fh (values %{$self->{files}}) {
00074     print $fh "<genomes>";
00075   }
00076   return;
00077 }

32 package Bio::EnsEMBL::MetaData::MetaDataDumper::XMLMetaDataDumper;
33 use base qw( Bio::EnsEMBL::MetaData::MetaDataDumper );
34 use Bio::EnsEMBL::Utils::Argument qw(rearrange);
35 use Carp;
36 use XML::Simple;
37 use strict;
38 use warnings;
39 
40 sub new {
41  my ($proto, @args) = @_;
42  my $self = $proto->SUPER::new(@args);
43  $self->{file} ||= "species_metadata.xml";
44  $self->{division} ||= 1;
45  return $self;
46 }
47 
48 sub do_dump {
49  my ($self, $metadata, $outfile) = @_;
50  $self->logger()->info("Writing XML to " . $outfile);
51  open(my $xml_file, '>', $outfile) ||
52  croak "Could not write to " . $outfile;
53  print $xml_file XML::Simple->new()
54  ->XMLout($self->metadata_to_hash($metadata), RootName => 'genomes');
55  close $xml_file;
56  $self->logger()->info("Completed writing XML to " . $outfile);
57  return;
58 }
59 
60 sub start {
61  my ($self, $divisions, $file, $dump_all) = @_;
62  $self->SUPER::start($divisions, $file, $dump_all);
63  for my $fh (values %{$self->{files}}) {
64  print $fh "<genomes>";
65  }
66  return;
67 }
68 
69 sub _write_metadata_to_file {
70  my ($self, $md, $fh, $count) = @_;
71  print $fh XML::Simple->new()
72  ->XMLout($md->to_hash(1), RootName => 'genome')
73  ;
74  return;
75 }
76 
77 sub end {
78  my ($self) = @_;
79  for my $fh (values %{$self->{files}}) {
80  print $fh "</genomes>\n";
81  }
82  $self->SUPER::end();
83  return;
84 }

00041 sub new {
00042   my ( $proto, @args ) = @_;
00043   my $self = $proto->SUPER::new(@args);
00044   ( $self->{regfile},  $self->{dbname}, $self->{pattern},
00045     $self->{division}, $self->{species} )
00046     = rearrange(
00047                  [ 'REGISTRY', 'DBNAME', 'PATTERN', 'DIVISION',
00048                    'SPECIES' ],
00049                  @args );
00050   $self->{registry} ||= 'Bio::EnsEMBL::Registry';
00051   $self->{logger} = get_logger();
00052   if ( defined $self->{regfile} ) {
00053       $self->{logger}->info("Loading from registry file ".$self->{regfile});
00054     $self->{registry}->load_all( $self->{regfile} );
00055   } else {
00056       $self->{logger}->info("Loading registry from arguments");
00057       $self->registry()->load_registry_from_db(@args);
00058       $self->registry()->set_disconnect_when_inactive(1);
00059   }
00060 
00061   return $self;

00069 sub get_dbas {
00070   my ($self) = @_;
00071   if ( !defined $self->{dbas} ) {
00072     $self->{dbas} = Bio::EnsEMBL::Registry->get_all_DBAdaptors();
00073     $self->{logger}
00074       ->info( "Found " . scalar( @{ $self->{dbas} } ) . " DBAs" );
00075     if ( defined $self->{dbname} ) {
00076       $self->{logger}
00077         ->info( "Restricting DBAs to DBs " . $self->{dbname} );
00078       $self->{dbas} = [ grep { $_->dbc()->dbname() eq $self->{dbname} }
00079                         @{ $self->{dbas} } ];
00080     }
00081     elsif ( defined $self->{pattern} ) {
00082       $self->{logger}
00083         ->info( "Restricting DBAs to DBs " . $self->{pattern} );
00084 
00085       $self->{dbas} = [
00086         grep {
00087           $_->dbc()->dbname() =~ m/$self->{pattern}/i
00088         } @{ $self->{dbas} } ];
00089     }
00090     if ( defined $self->{species} ) {
00091       $self->{logger}
00092         ->info( "Restricting DBAs to species " . $self->{species} );
00093       $self->{dbas} = [ grep { $_->species() eq $self->{species} }
00094                         @{ $self->{dbas} } ];
00095     }
00096     elsif ( defined $self->{division} ) {
00097       $self->{logger}
00098         ->info( "Restricting DBAs to division " . $self->{division} );
00099       $self->{dbas} = [
00100         grep {
00101           ref( $_->get_MetaContainer() ) eq
00102             'Bio::EnsEMBL::DBSQL::MetaContainer' &&
00103             $_->get_MetaContainer()->get_division() =~
00104             m/$self->{division}/i
00105         } @{ $self->{dbas} } ];
00106     }
00107     $self->{logger}
00108       ->info( "Filtered to " . scalar( @{ $self->{dbas} } ) . " DBAs" );
00109   } ## end if ( !defined $self->{...})
00110   return $self->{dbas};
00111 } ## end sub get_dbas

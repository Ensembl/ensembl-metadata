22 =head1 NAME
23 
24 Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor
25 
26 =head1 SYNOPSIS
27 
28 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->build_adaptor();
29 my $md = $gdba->fetch_by_name("arabidopsis_thaliana");
30 
31 =head1 DESCRIPTION
32 
33 Adaptor for storing and retrieving GenomeInfo objects from MySQL genome_info database
34 
35 To start working with an adaptor:
36 
37 # getting an adaptor
38 ## adaptor for latest public EG release
39 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->build_eg_adaptor();
40 ## adaptor for specified public EG release
41 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->build_eg_adaptor(21);
42 ## manually specify a given database
43 my $dbc = Bio::EnsEMBL::DBSQL::DBConnection->new(
44 -USER=>'anonymous',
45 -PORT=>4157,
46 -HOST=>'mysql-eg-publicsql.ebi.ac.uk',
47 -DBNAME=>'genome_info_21');
48 my $gdba = Bio::EnsEMBL::MetaData::DBSQL::GenomeInfoAdaptor->new(-DBC=>$dbc);
49 
50 To find genomes, use the fetch methods e.g.
51 
52 # find a genome by name
53 my $genome = $gdba->fetch_by_name('arabidopsis_thaliana');
54 
55 # find and iterate over all genomes
56 for my $genome (@{$gdba->fetch_all()}) {
57  print $genome->name()."\n";
58 }
59 
60 # find and iterate over all genomes from plants
61 for my $genome (@{$gdba->fetch_all_by_division('EnsemblPlants')}) {
62  print $genome->name()."\n";
63 }
64 
65 # find and iterate over all genomes with variation
66 for my $genome (@{$gdba->fetch_all_with_variation()}) {
67  print $genome->name()."\n";
68 }
69 
70 # find all comparas for the division of interest
71 my $comparas = $gdba->fetch_all_compara_by_division('EnsemblPlants');
72 
73 # find the peptide compara
74 my ($compara) = grep {$_->is_peptide_compara()} @$comparas;
75 print $compara->division()." ".$compara->method()."(".$compara->dbname().")\n";
76 
77 # print out all the genomes in this compara
78 for my $genome (@{$compara->genomes()}) {
79  print $genome->name()."\n";
80 }

102 sub store {
103  my ( $self, $data_release ) = @_;
104  if ( !defined $data_release->dbID() ) {
105  # find out if organism exists first
106  my $dbID;
107  if ( defined $data_release->ensembl_genomes_version() ) {
108  ($dbID) =
109  @{
110  $self->dbc()->sql_helper()->execute_simple(
111  -SQL =>
112 "select data_release_id from data_release where ensembl_version=? and ensembl_genomes_version=?",
113  -PARAMS => [ $data_release->ensembl_version(),
114  $data_release->ensembl_genomes_version() ] ) };
115  }
116  else {
117  ($dbID) =
118  @{
119  $self->dbc()->sql_helper()->execute_simple(
120  -SQL =>
121 "select data_release_id from data_release where ensembl_version=? and ensembl_genomes_version is null",
122  -PARAMS => [ $data_release->ensembl_version() ] ) };
123  }
124 
125  if ( defined $dbID ) {
126  $data_release->dbID($dbID);
127  $data_release->adaptor($self);
128  }
129  } ## end if ( !defined $data_release...)
130  if ( defined $data_release->dbID() ) {
131  $self->update($data_release);
132  }
133  else {
134  $self->dbc()->sql_helper()->execute_update(
135  -SQL =>
136 q/insert into data_release(ensembl_version,ensembl_genomes_version,release_date,is_current) values (?,?,?,?)/,
137  -PARAMS => [ $data_release->ensembl_version(),
138  $data_release->ensembl_genomes_version(),
139  $data_release->release_date() ],
140  -CALLBACK => sub {
141  my ( $sth, $dbh, $rv ) = @_;
142  $data_release->dbID( $dbh->{mysql_insertid} );
143  } );
144  $data_release->adaptor($self);
145  $self->_store_databases($data_release);
146  $self->_store_cached_obj($data_release);
147  }
148  return;
149 } ## end sub store

151 sub update {
152  my ( $self, $data_release ) = @_;
153  if ( !defined $data_release->dbID() ) {
154  croak "Cannot update an object that has not already been stored";
155  }
156 
157  $self->dbc()->sql_helper()->execute_update(
158  -SQL =>
159 q/update data_release set ensembl_version=?, ensembl_genomes_version=?, release_date=? where data_release_id=?/,
160  -PARAMS => [ $data_release->ensembl_version(),
161  $data_release->ensembl_genomes_version(),
162  $data_release->release_date(),
163  $data_release->dbID() ] );
164 
165  $self->_store_databases($data_release);
166  return;
167 }

207 sub fetch_databases {
208  my ( $self, $release, $division ) = @_;
209  my $databases = $release->databases();
210  if(defined $division) {
211  $databases = [grep {$division eq $_->division()} @{$release->databases()}];
212  }
213  if ( defined $division ) {
214  $databases = [ @$databases,
215  @{$self->db()->get_GenomeInfoAdaptor()
216  ->fetch_division_databases( $division, $release ) },
217  @{$self->db()->get_GenomeInfoAdaptor()
218  ->fetch_division_databases( $division, $release ) } ];
219  }
220  else {
221  $databases = [ @$databases,
222  @{$self->db()->get_GenomeInfoAdaptor()
223  ->fetch_databases($release) },
224  @{$self->db()->get_GenomeInfoAdaptor()
225  ->fetch_databases($release) } ];
226  }
227  return $databases;
228 } ## end sub fetch_databases

254 sub _fetch_databases {
255  my ( $self, $release ) = @_;
256  croak
257 "Cannot fetch databases for a DataReleaseInfo object that has not been stored"
258  if !defined $release->dbID();
259  my $databases =
260  $self->db()->get_DatabaseInfoAdaptor()
261  ->fetch_databases($release);
262  $release->databases($databases);
263  return;
264 }

275 sub _store_databases {
276  my ( $self, $release ) = @_;
277  $self->dbc()->sql_helper()->execute_update(
278  -SQL => q/delete from data_release_database where data_release_id=?/,
279  -PARAMS => [ $release->dbID() ] );
280 
281  for my $database ( @{ $release->databases() } ) {
282  print "Storing database ".$database->dbname()."\n";
283  $self->db()->get_DatabaseInfoAdaptor()->store($database);
284  }
285  return;
286 }

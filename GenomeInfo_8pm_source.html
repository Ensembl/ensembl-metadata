00036 =head1 DESCRIPTION
00037 
00038 Object encapsulating meta information about a genome in Ensembl Genomes. 
00039 
00040 Can be used to render information about a genome e.g.
00041 
00042 print $genome->name()." (".$genome->species.")\n";
00043 print "Sequences: ".scalar(@{$genome->sequences()})."\n";
00044 if($genome->has_variations()) {
00045     print "Variations: \n";
00046     # variations is a hash with type as the key
00047     while(my ($type,$value) = each %{$genome->variations()}) {
00048         print "- $type\n";
00049     }
00050 }
00051 print "Compara analyses: ".scalar(@{$genome->compara()})."\n";

00796 =head2 to_hash
00797   Description: Render genome as plain hash suitable for export as JSON/XML
00798   Argument   : (optional) if set to 1, force expansion of children
00799   Returntype : Hashref
00800   Exceptions : none
00801   Caller     : general
00802   Status     : Stable
00803 =cut
00804 
00805 sub to_hash {
00806   my ( $in, $keen ) = @_;
00807   my $out;
00808   my $type = ref $in;
00809   if ( defined $keen &&
00810        $keen == 1 &&
00811        $type =~ m/Bio::EnsEMBL::MetaData::[A-z]+Info/ )
00812   {
00813     $in->_preload();
00814   }
00815   if ( $type eq 'ARRAY' ) {
00816     $out = [];
00817     for my $item ( @{$in} ) {
00818       push @{$out}, to_hash( $item, $keen );
00819     }
00820   }
00821   elsif ( $type eq 'HASH' || $type eq 'Bio::EnsEMBL::MetaData::GenomeInfo' ) {
00822     $out = {};
00823     while ( my ( $key, $val ) = each %$in ) {
00824       if ( $key ne 'dbID' && $key ne 'adaptor' && $key ne 'logger' ) {
00825 
00826 # deal with keys starting with numbers, which are not valid element names in XML
00827         if ( $key =~ m/^[0-9].*/ ) {
00828           $key = '_' . $key;
00829         }
00830         $out->{$key} = to_hash( $val, $keen );
00831       }
00832 
00833     }
00834   }
00835   elsif ( $type =~ m/Bio::EnsEMBL::MetaData::[A-z]+Info/ ) {
00836     $out = $in->to_hash($keen);
00837   }
00838   else {
00839     $out = $in;
00840   }
00841   if ( defined $keen &&
00842        $keen == 1 &&
00843        $type =~ m/Bio::EnsEMBL::MetaData::[A-z]+Info/ )
00844   {
00845     $in->_preload();
00846   }
00847   return $out;
00848 } ## end sub to_hash

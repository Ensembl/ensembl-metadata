20 use warnings;
21 use strict;
22 
23 package Bio::EnsEMBL::MetaData::Pipeline::GenomeFactory;
24 
25 use base qw/Bio::EnsEMBL::Production::Pipeline::BaseSpeciesFactory/;
26 
27 use Bio::EnsEMBL::MetaData::MetaDataProcessor;
28 use Bio::EnsEMBL::MetaData::AnnotationAnalyzer;
29 use Bio::EnsEMBL::MetaData::DBSQL::MetaDataDBAdaptor;
30 use Bio::EnsEMBL::MetaData::DataReleaseInfo;
31 
32 use Carp;
33 use Log::Log4perl qw(:easy);
34 Log::Log4perl->easy_init($INFO);
35 my $log = get_logger();
36 
37 sub run {
38  my ($self) = @_;
39 
40  # create a release to use
41  $log->info("Connecting to info database");
42  my $dba =
43  Bio::EnsEMBL::MetaData::DBSQL::MetaDataDBAdaptor->new(
44  -USER =>,
45  $self->param('info_user'),
46  -PASS =>,
47  $self->param('info_pass'),
48  -HOST =>,
49  $self->param('info_host'),
50  -PORT =>,
51  $self->param('info_port'),
52  -DBNAME =>,
53  $self->param('info_dbname')
54  );
55 
56  my $rdba = $dba->get_DataReleaseInfoAdaptor();
57  my $release = $self->param('release');
58  my $eg_release = $self->param('eg_release');
59  my $release_date = $self->param('release_date');
60  $log->info( "Storing release e$release" . ( ( defined $eg_release ) ?
61  "/EG$eg_release" : "" ) .
62  " $release_date" );
63 
64  $rdba->store(
65  Bio::EnsEMBL::MetaData::DataReleaseInfo->new(
66  -ENSEMBL_VERSION => $release,
67  -ENSEMBL_GENOMES_VERSION => $eg_release,
68  -RELEASE_DATE => $release_date )
69  );
70 
71  $log->info("Completed release creation");
72 
73  # run rest of process
74  $self->SUPER::run();
75  return;
76 } ## end sub run
77 
78 1;

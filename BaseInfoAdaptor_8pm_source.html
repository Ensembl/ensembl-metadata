00141 sub _args_to_sql {
00142   my ( $self, $sql_in, $args ) = @_;
00143   my $sql    = $sql_in;
00144   my $params = [];
00145   my $clause = '';
00146   while ( my ( $k, $v ) = each %$args ) {
00147     if ( $clause ne '' ) {
00148       $clause .= ' AND ';
00149     }
00150     if ( ref($v) eq 'ARRAY' ) {
00151       $clause .= "$k in (" . join( ',', map { '?' } @$v ) . ")";
00152       $params = [ @$params, @$v ];
00153     }
00154     else {
00155       $clause .= "$k = ?";
00156       push @$params, $v;
00157     }
00158   }
00159   if ( $clause ne '' ) {
00160     $sql .= ' where ' . $clause;
00161   }
00162   return ( $sql, $params );

00176 sub _fetch_generic {
00177   my ( $self, $sql, $params, $type, $keen ) = @_;
00178   if ( !defined $type ) {
00179     $type = $self->_get_obj_class();
00180   }
00181   my $mds = $self->{dbc}->sql_helper()->execute(
00182     -SQL          => $sql,
00183     -USE_HASHREFS => 1,
00184     -CALLBACK     => sub {
00185       my $row = shift @_;
00186       my $md = $self->_get_cached_obj( $type, $row->{dbID} );
00187       if ( !defined $md ) {
00188         $md = bless $row, $type;
00189         $md->adaptor($self);
00190         $self->_store_cached_obj($md);
00191       }
00192       return $md;
00193     },
00194     -PARAMS => $params );
00195   if ($keen) {
00196     for my $md ( @{$mds} ) {
00197       $self->_fetch_children( $md, $keen );
00198     }
00199   }
00200   return $mds;

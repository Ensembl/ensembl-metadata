00182 sub _args_to_sql {
00183   my ( $self, $sql_in, $args ) = @_;
00184   my $sql    = $sql_in;
00185   my $params = [];
00186   my $clause = '';
00187   while ( my ( $k, $v ) = each %$args ) {
00188     if ( $clause ne '' ) {
00189       $clause .= ' AND ';
00190     }
00191     if ( ref($v) eq 'ARRAY' ) {
00192       $clause .= "$k in (" . join( ',', map { '?' } @$v ) . ")";
00193       $params = [ @$params, @$v ];
00194     }
00195     else {
00196       $clause .= "$k = ?";
00197       push @$params, $v;
00198     }
00199   }
00200   if ( $clause ne '' ) {
00201     $sql .= ' where ' . $clause;
00202   }
00203   return ( $sql, $params );

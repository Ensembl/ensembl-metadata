91 sub fetch_by_dbIDs {
92  my ( $self, $ids, $keen ) = @_;
93  my @genomes = ();
94  my $it = natatime 1000, @{$ids};
95  while ( my @vals = $it->() ) {
96  my $sql =
97  $self->_get_base_sql() . ' where ' . $self->_get_id_field() . ' in (' .
98  join( ',', @vals ) . ')';
99  @genomes = ( @genomes, @{ $self->_fetch_generic( $sql, [] ) } );
100  }
101  return \@genomes;

141 sub _args_to_sql {
142  my ( $self, $sql_in, $args ) = @_;
143  my $sql = $sql_in;
144  my $params = [];
145  my $clause = '';
146  while ( my ( $k, $v ) = each %$args ) {
147  if ( $clause ne '' ) {
148  $clause .= ' AND ';
149  }
150  if ( ref($v) eq 'ARRAY' ) {
151  $clause .= "$k in (" . join( ',', map { '?' } @$v ) . ")";
152  $params = [ @$params, @$v ];
153  }
154  else {
155  $clause .= "$k = ?";
156  push @$params, $v;
157  }
158  }
159  if ( $clause ne '' ) {
160  $sql .= ' where ' . $clause;
161  }
162  return ( $sql, $params );
163 }

166  Arg : SQL to use to fetch object
167  Arg : arrayref of bind parameters
168  Arg : (optional) if set to 1, all children will be fetched
169  Description: Instantiate a GenomeInfo from the database using the specified SQL

173  Status : Stable

176 sub _fetch_generic {
177  my ( $self, $sql, $params, $type, $keen ) = @_;
178  if ( !defined $type ) {
179  $type = $self->_get_obj_class();
180  }
181  my $mds = $self->{dbc}->sql_helper()->execute(
182  -SQL => $sql,
183  -USE_HASHREFS => 1,
184  -CALLBACK => sub {
185  my $row = shift @_;
186  my $md = $self->_get_cached_obj( $type, $row->{dbID} );
187  if ( !defined $md ) {
188  $md = bless $row, $type;
189  $md->adaptor($self);
190  $self->_store_cached_obj($md);
191  }
192  return $md;
193  },
194  -PARAMS => $params );
195  if ($keen) {
196  for my $md ( @{$mds} ) {
197  $self->_fetch_children( $md, $keen );
198  }
199  }
200  return $mds;
201 } ## end sub _fetch_generic

212 my $division_names = { 'Ensembl' => 'E',
213  'EnsemblGenomes' => 'EG',
214  'EnsemblBacteria' => 'EB',
215  'EnsemblMetazoa' => 'EM',
216  'EnsemblPlants' => 'EPl',

220 sub _get_division_id {
221  my ( $self, $division, $short_name ) = @_;
222 
223  my $div_id = $self->_cache('division')->{$division};
224 
225  if ( !defined $div_id ) {
226 
227  $div_id = $self->{dbc}->sql_helper()->transaction(
228  -CALLBACK => sub {
229  my ($dbc) = @_;
230  my $id;
231  my $ids =
232  $dbc->sql_helper()->execute_simple(
233  -SQL => 'select division_id from division where name=?',
234  -PARAMS => [$division] );
235  if ( scalar(@$ids) == 0 ) {
236  # not found so create
237  if ( !defined $short_name ) {
238  $short_name = $division_names->{$division};
239  if ( !defined $short_name ) {
240  $short_name = $division;
241  $short_name =~ s/[a-z]+//g;
242  }
243  }
244  $dbc->sql_helper->execute_update(
245  -SQL => 'insert into division (name,short_name) values(?,?)',
246  -CALLBACK => sub {
247  my ( $sth, $dbh, $rv ) = @_;
248  $id = $dbh->{mysql_insertid};
249  },
250  -PARAMS => [ $division, $short_name ] );
251  }
252  else {
253  $id = $ids->[0];
254  }
255  return $id;
256  } );
257  $self->_cache('division')->{$division} = $div_id;
258  } ## end if ( !defined $div_id )
259 
260  return $div_id;
261 } ## end sub _get_division_id
